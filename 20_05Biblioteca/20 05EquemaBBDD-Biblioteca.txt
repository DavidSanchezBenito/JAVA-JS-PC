CREATE DATABASE biblioteca;
use biblioteca;
CREATE TABLE COPIA (ID_ELMENTO        INT, NUM_COPIA        INT, PRESTABLE        BIT);
CREATE TABLE ENPRESTAMO (ID_USUARIO INT, ID_ELEMENTO INT, NUM_COPIA INT, NUM_PRESTAMOS INT, FECHA_ENTREGA DATE);
CREATE TABLE ELEMENTO (ID_ELEMENTO INT, TIPO STRING, ISBN STRING, TITULO STRING, AUTOR STRING, LIMITE STRING, VOLUMENES INT, FECHA_PUBLI DATE);
CREATE TABLE MAXTABLA (ID_TABLA INT, ID_CLAVE INT, MAXIMO INT);
CREATE TABLE USUARIO (ID_USUARIO INT, NOMBRE STRING, APELLIDOS STRING, PASSWORD STRING, CORREOE STRING);
